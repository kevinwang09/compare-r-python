Creating a data frame manually

R

data = data.frame(
  x = 1:5, 
  y = 2:6,
  row.names = c("a", "b", "c", "d", "e"))
print(data)
##   x y
## a 1 2
## b 2 3
## c 3 4
## d 4 5
## e 5 6

Python

import pandas as pd
data = pd.DataFrame(
  {"x": list(range(1, 6)),
   "y": list(range(2, 7))}, 
  index = ["a", "b", "c", "d", "e"])
print(data)
##    x  y
## a  1  2
## b  2  3
## c  3  4
## d  4  5
## e  5  6
